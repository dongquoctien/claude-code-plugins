# NgRx feature store — required layout

Every feature module generated by this plugin MUST include an NgRx feature store. Layout matches the real `routes/<feature>/` shape in oh-admin and partners.

## Folder layout (mandatory)

```
routes/<feature>/
├── <feature>.module.ts            // imports StoreModule.forFeature + EffectsModule.forFeature
├── <feature>-routing.module.ts    // single-child route, container component
├── actions/
│   ├── <feature>.actions.ts       // every createAction declaration
│   └── index.ts                   // `import * as ThisActions from './<feature>.actions'; export { ThisActions };`
├── reducers/
│   ├── <feature>.reducer.ts       // featureKey, State, adapter, initialState, reducer fn, named selectors
│   └── index.ts                   // outer featureKey, FeatureState shape, combineReducers, top-level selectors
├── effects/
│   ├── <feature>.effects.ts       // class with createEffect blocks
│   └── index.ts                   // `export * from './<feature>.effects';` so module import is mechanical
├── services/
│   ├── <feature>-api.service.ts   // wraps ApiService.post/get, validates succeedYn
│   ├── <feature>-data.service.ts  // optional — derived data utilities
│   └── index.ts
├── models/
│   ├── <feature>.model.ts         // I-interfaces for state, request, response
│   ├── api-path.model.ts          // BsEventApiPath enum-like with URLs
│   └── index.ts
├── containers/
│   ├── <feature>.component.{ts,html,css}    // smart component bound to store
│   └── index.ts
└── components/
    ├── <feature>-search.component.{ts,html,css}
    ├── <feature>-form.component.{ts,html,css}
    └── index.ts
```

## `actions/<feature>.actions.ts` pattern

```typescript
import { createAction, props } from '@ngrx/store';
import { I<Feature>Request, I<Feature>ResponseItem } from '../models';

export const initComponent = createAction('[<Feature>] init this component');

//#region search / list
export const loadList = createAction(
  '[<Feature>] load list',
  props<{ condition: I<Feature>Request }>(),
);
export const getListSuccess = createAction(
  '[<Feature>] get list success',
  props<{ dataList: I<Feature>ResponseItem[]; totalCount: number }>(),
);
export const getListFailure = createAction(
  '[<Feature>] get list failure',
  props<{ error: any }>(),
);
//#endregion

//#region grid state
export const setGridStateAction = createAction(
  '[<Feature>] set grid state',
  props<{ gridState: any }>(),
);
//#endregion
```

**Action name convention**: `'[<PascalFeature>] <verb-phrase>'` — square-bracketed source, then lowercase verb phrase. Triples: `loadX` → `getXSuccess` → `getXFailure`.

## `reducers/<feature>.reducer.ts` pattern

```typescript
import { on, createReducer, Action } from '@ngrx/store';
import { createEntityAdapter, EntityAdapter, EntityState } from '@ngrx/entity';

import { I<Feature>ResponseItem } from '../models';
import { ThisActions } from '../actions';

export const featureKey = '<featureCamel>List';

export interface State extends EntityState<I<Feature>ResponseItem> {
  loading: boolean;
  searchCondition: any;
  gridState: any;
  gridDataTotal: number;
  // ...feature-specific slices...
}

export const adapter: EntityAdapter<I<Feature>ResponseItem> = createEntityAdapter<I<Feature>ResponseItem>({
  selectId: (item) => item.rowid,
  sortComparer: false,
});

const initialState: State = adapter.getInitialState({
  loading: false,
  searchCondition: null,
  gridState: { skip: 0, take: 50, sort: [], filter: { logic: 'and', filters: [] } },
  gridDataTotal: 0,
});

const featureReducer = createReducer(
  initialState,
  on(ThisActions.loadList, (state) => ({ ...state, loading: true })),
  on(ThisActions.getListSuccess, (state, { dataList, totalCount }) =>
    adapter.setAll(dataList, { ...state, loading: false, gridDataTotal: totalCount }),
  ),
  on(ThisActions.getListFailure, (state) => ({ ...state, loading: false })),
);

export function reducer(state: State | undefined, action: Action) {
  return featureReducer(state, action);
}

// ─── local selectors (used by reducers/index.ts) ───
export const selectLoading = (state: State) => state.loading;
export const selectGridState = (state: State) => state.gridState;
export const selectGridDataTotal = (state: State) => state.gridDataTotal;
export const { selectAll: selectAllItems } = adapter.getSelectors();
```

## `reducers/index.ts` pattern (the wrapper)

```typescript
import { createSelector, createFeatureSelector, combineReducers, Action } from '@ngrx/store';
import * as fromList from './<feature>.reducer';
import * as fromRoot from 'src/app/reducers';

export const featureKey = '<featureCamel>';

export interface <Feature>State {
  [fromList.featureKey]: fromList.State;
}

export interface State extends fromRoot.State {
  [featureKey]: <Feature>State;
}

export function reducers(state: <Feature>State | undefined, action: Action) {
  return combineReducers({
    [fromList.featureKey]: fromList.reducer,
  })(state, action);
}

// ─── top-level selectors ──────────────────────────────────────────────
export const selectModuleState = createFeatureSelector<State, <Feature>State>(featureKey);
export const selectListState = createSelector(
  selectModuleState,
  (state) => state[fromList.featureKey],
);

export const selectLoading = createSelector(selectListState, fromList.selectLoading);
export const selectGridState = createSelector(selectListState, fromList.selectGridState);
export const selectGridDataTotal = createSelector(selectListState, fromList.selectGridDataTotal);
export const selectAllItems = createSelector(selectListState, fromList.selectAllItems);
```

Note the **two-level featureKey**: outer `<feature>` for the module slice, inner `<feature>List` (or `<feature>Form`, `<feature>Detail`) for each sub-reducer. This is the project pattern; do NOT flatten to one level.

## `effects/<feature>.effects.ts` pattern

```typescript
import { Injectable } from '@angular/core';
import { Observable, of } from 'rxjs';
import { catchError, concatMap, map, switchMap, withLatestFrom } from 'rxjs/operators';
import { Store, select } from '@ngrx/store';
import { Actions, createEffect, ofType } from '@ngrx/effects';

import { ThisActions } from '../actions';
import * as fromModule from '../reducers';
import { <Feature>ApiService } from '../services';
import { ErrorHandlerService } from 'src/app/core/services/error/error-handler.service';

@Injectable()
export class <Feature>Effects {
  constructor(
    private actions$: Actions,
    private service: <Feature>ApiService,
    private _errorService: ErrorHandlerService,
    private store: Store<fromModule.State>,
  ) {}

  loadList$ = createEffect(() =>
    this.actions$.pipe(
      ofType(ThisActions.loadList),
      switchMap(({ condition }) =>
        this.service.getList(condition).pipe(
          map(({ dataList, totalCount }) => ThisActions.getListSuccess({ dataList, totalCount })),
          catchError((error) => {
            this._errorService.handleError(error);
            return of(ThisActions.getListFailure({ error }));
          }),
        ),
      ),
    ),
  );
}
```

## `services/<feature>-api.service.ts` pattern

```typescript
import { Injectable } from '@angular/core';
import { Observable, of, throwError } from 'rxjs';
import { switchMap } from 'rxjs/operators';
import * as _ from 'lodash';

import { ApiService } from 'src/app/core/services/api/api.service';
import { <Feature>ApiPath as ApiPath } from '../models';

@Injectable({ providedIn: 'root' })
export class <Feature>ApiService {
  constructor(private api: ApiService) {}

  getList(condition: any): Observable<{ dataList: any[]; totalCount: number }> {
    const params = _.assign({}, this.api.defaultParams, { condition });
    return this.api.post(ApiPath.getList, params).pipe(
      switchMap((res) => {
        if (res.succeedYn && res.result) {
          return of({ dataList: this._addRowid(res.result.list), totalCount: res.result.totalCount });
        } else {
          return throwError(res || 'api response error');
        }
      }),
    );
  }

  private _addRowid(list: any[]): any[] {
    return (list || []).map((item, idx) => ({ ...item, rowid: idx + 1 }));
  }
}
```

**API path enum** lives in `models/api-path.model.ts`:
```typescript
export enum <Feature>ApiPath {
  getList = '/admin/<resource>/list',
  getDetail = '/admin/<resource>/detail',
}
```

## Module imports — what's mandatory

```typescript
@NgModule({
  imports: [
    CommonModule,
    FormsModule,
    ReactiveFormsModule,

    // Kendo where applicable
    GridModule, InputsModule, KendoCommonModule,

    // Shared (from reuse-map)
    PipesModule, DialogModule, DropdownModule, SelectboxModule, CheckboxModule,
    InputTextModule, AutoCompleteModule, DatePickerModule, GridSimpleModule,
    DirectivesModule,

    // Feature routing
    <Feature>RoutingModule,

    // NgRx — MANDATORY
    StoreModule.forFeature(fromModule.featureKey, fromModule.reducers),
    EffectsModule.forFeature([<Feature>Effects]),
  ],
  declarations: [CONTAINERS, COMPONENTS],
})
export class <Feature>Module {}
```

## Route data — what's mandatory

```typescript
{
  path: '',
  component: <Feature>Component,
  data: { title: '<Human-readable title>', store: '<featureCamel>' },
}
```

The `data.store` field is consumed by `deleteMenuTabReducer` to clean state when tabs close. Skip it and tab cleanup breaks.

## Effect-error pattern reminder

Every effect MUST `catchError` and dispatch a `*Failure` action; service-level `throwError` is what triggers it. Never let an error bubble past the effect — it kills the action stream.
